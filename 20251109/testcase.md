- [ ] 空文字列を含む場合
    - [ ] 複数の文字列と空文字列を含む場合に適切にハッシュを作成できる
        - [ ] ["test", "", "data"] => {"T"=>["test"], "D"=>["data"]}
    - [ ] 空文字列やから配列の場合に適切にハッシュを作成できる
        - [ ] [""] => {}
        - [ ] [] => {}

- [ ] 空文字列を含まない場合
    - [ ] 小文字のみの場合でも適切にハッシュを作成できる
        - [ ] ["apple", "banana", "avocado", "cherry"] => {"A"=>["apple", "avocado"], "B"=>["banana"], "C"=>["cherry"]}
    - [ ] 大文字を含む場合にも適切にハッシュを作成できる
        - [ ] ["Ruby", "rails", "React", "redis"] => {"R"=>["Ruby", "rails", "React", "redis"]}
